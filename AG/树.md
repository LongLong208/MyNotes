#### 后序遍历

迭代写法:

```cpp {cmd=run}
//sf
vector<int> postorderTraversal(TreeNode<int>* root) {
    vector<int> ans;
    if (root == nullptr)
        return ans;
    
    stack<TreeNode<int>*> stk;
    TreeNode<int>* prev = nullptr;
    while (root != nullptr || !stk.empty()) {
        while (root != nullptr) {
            stk.push(root);
            root = root->left;
        }
        root = stk.top();
        stk.pop();
        if (root->right == nullptr || root->right == prev) {
            ans.push_back(root->val);
            prev = root;
            root = nullptr;
        } else {
            stk.push(root);
            root = root->right;
        }
    }
    return ans;
}
```
